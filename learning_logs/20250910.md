class Person:

    population = 0

    def __init__(self, name):
        self.name = name
        Person.population += 1
        print(f'{self.name}이(가) 태어남.')

    def __del__(self):
        print(f'RIP {self.name}')
        Person.population -= 1

    @classmethod
    def get_population(cls):
        return cls.population

man = Person('김일')
woman = Person('김이')
print(f'전체 인구수 : {Person.get_population()}\n')

del man
print(f'전체 인구수 : {Person.get_population()}')
